import sys
import heapq

N, K = list(map(int, sys.stdin.readline().split()))
use = list(map(int, sys.stdin.readline().split()))


def solution():
    multitab = []
    ans = 0

    for idx, u in enumerate(use):
        if u in multitab:
            continue

        if len(multitab) < N:
            multitab.append(u)
        else:
            pq = []
            for tab in multitab:
                first = K
                for w in range(idx, K):
                    if use[w] == tab:
                        first = w
                        break
                heapq.heappush(pq, (-first, tab))
            pos, target = heapq.heappop(pq)
            for i in range(N):
                if multitab[i] == target:
                    multitab[i] = u
                    break
            ans += 1

    return ans


print(solution())
