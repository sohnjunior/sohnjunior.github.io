/**
 * @param {number[]} nums
 * @return {number}
 */
var findDuplicate = function (nums) {
  let lo = 1;
  let hi = nums.length - 1;

  let ans = 0;
  while (lo <= hi) {
    let mid = Math.floor((lo + hi) / 2);

    if (count(nums, mid) > mid) {
      ans = mid;
      hi = mid - 1;
    } else {
      lo = mid + 1;
    }
  }

  return ans;
};

function count(nums, pivot) {
  return nums.filter((e) => e <= pivot).length;
}
