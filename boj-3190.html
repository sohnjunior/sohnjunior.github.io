    line = list(sys.stdin.readline().split())
    rotate.append([int(line[0]), line[1]])


state = {
    'blank': 0,
    'wall': 1,
    'body': 2,
    'apple': 3,

    tail_dir = 'e'

        board[i][N+1] = 1

import sys
from collections import deque


N = int(input())
K = int(input())
apples = []
for _ in range(K):
    apples.append(list(map(int, sys.stdin.readline().split())))
L = int(input())
rotate = []
for _ in range(L):
    line = sys.stdin.readline().split()
    rotate.append([int(line[0]), line[1]])


next_dir = [
    [2, 3],
    [3, 2],
    [1, 0],
    [0, 1]
]

dx = [-1, 1, 0, 0]
dy = [0, 0, -1, 1]


def solution():
    global rotate

    time = 0
    snake = deque()
    snake.append([0, 0, 3])  # (0, 0) ì§€ì ì—ì„œ ì˜¤ë¥¸ìª½(3) ìœ¼ë¡œ ì‹œì‘

    board = [[0]*N for _ in range(N)]
    board[0][0] = 1

    # ì§€ë„ì— ì‚¬ê³¼ í‘œì‹œ
    for x, y in apples:
        board[x-1][y-1] = 2

    # ë±€ ê²Œì„ ì‹œë®¬ë ˆì´ì…˜
    while True:
        hx, hy, dir = snake[0]

        # íšŒì „í•  ì‹œê°„ì´ ë˜ë©´
        if rotate and time == rotate[0][0]:
            if rotate[0][1] == 'L':
                dir = next_dir[dir][0]
            else:
                dir = next_dir[dir][1]
            rotate = rotate[1:]

        # ë²½ ë˜ëŠ” ìê¸°ìì‹ ì˜ ëª¸ì— ë¶€ë”ªíˆë©´
        nx, ny = hx + dx[dir], hy + dy[dir]
        if nx < 0 or nx >= N or ny < 0 or ny >= N or board[nx][ny] == 1:
            time += 1
            break

        # ë‹¤ìŒ ìœ„ì¹˜ê°€ ì‚¬ê³¼ì´ë©´
        if board[nx][ny] == 2:
            board[nx][ny] = 1
            snake.appendleft([nx, ny, dir])
        else:
            board[nx][ny] = 1
            snake.appendleft([nx, ny, dir])
            board[snake[-1][0]][snake[-1][1]] = 0
            snake.pop()

        time += 1

    return time


print(solution())
