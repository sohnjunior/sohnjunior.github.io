import sys


def solution(bitA, bitB):
    a_cnt, b_cnt, diff = 0, 0, 0
    for i in range(len(bitA)):
        if bitA[i] == '1':
            a_cnt += 1
        if bitB[i] == '1':
            b_cnt += 1

        if bitA[i] != bitB[i]:
            diff += 1

    invert = abs(a_cnt - b_cnt)
    return invert + (diff - invert) // 2


if __name__ == '__main__':
    T = int(input())
    for _ in range(T):
        bits = list(sys.stdin.readline().split())
        answer = solution(*bits)
        print(answer)
