  const qry = "SELECT * FROM board";
  connection.query(qry, (err, rows) => {
    if (err) {
      console.error("query error" + err);
      res.status(500).send("Internal Server Error");

      res.render("index", { title: "MySQL ì—°ë™ Demo", rows: rows });

});
