import sys


K, N = list(map(int, sys.stdin.readline().split()))
lines = []
for _ in range(K):
    lines.append(int(input()))


def enough(cut):
    ret = 0
    for line in lines:
        ret += line // cut
    return True if ret >= N else False


def solution():
    lo = 1
    hi = max(lines)

    answer = 0
    while lo <= hi:
        mid = (lo + hi) // 2
        if enough(mid):
            answer = mid
            lo = mid + 1
        else:
            hi = mid - 1

    return answer


print(solution())
