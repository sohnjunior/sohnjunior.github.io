import sys
from collections import deque

LIMIT = 9999


def bfs(A, B):
    q = deque()
    visit = [0] * (LIMIT+1)

    visit[A] = 1
    q.append((A, ''))

    while q:
        here, command = q.popleft()
        if here == B:
            return command

        # D ì—°ì‚°
        there = 2 * here if 2 * here <= LIMIT else (2 * here) % (LIMIT + 1)
        if not visit[there]:
            q.append((there, command + 'D'))
            visit[there] = 1

        # S ì—°ì‚°
        there = here - 1 if here != 0 else LIMIT
        if not visit[there]:
            q.append((there, command + 'S'))
            visit[there] = 1

        # L ì—°ì‚°
        there = (here % 1000) * 10 + here // 1000
        if not visit[there]:
            q.append((there, command + 'L'))
            visit[there] = 1

        # R ì—°ì‚°
        there = (here % 10) * 1000 + here // 10
        if not visit[there]:
            q.append((there, command + 'R'))
            visit[there] = 1


if __name__ == '__main__':
    T = int(input())
    for _ in range(T):
        A, B = list(map(int, sys.stdin.readline().split()))
        ans = bfs(A, B)
        print(ans)
