import sys
import heapq


N, M, X = list(map(int, sys.stdin.readline().split()))
edges = []
for _ in range(M):
    edges.append(list(map(int, sys.stdin.readline().split())))


def dijkstra(adj):
    # ëª¨ë“  ì •ì ì—ì„œ ë‹¤ìµìŠ¤íŠ¸ë¼ ìˆ˜í–‰
    go_party = []
    go_back = []
    for start in range(1, N+1):
        dist = [sys.maxsize for _ in range(N+1)]
        pq = []
        heapq.heappush(pq, (0, start))
        dist[start] = 0

        while pq:
            cost, here = heapq.heappop(pq)

            if cost > dist[here]:
                continue

            for there in adj[here]:
                nextDist = there[1] + cost
                if dist[there[0]] > nextDist:
                    dist[there[0]] = nextDist
                    heapq.heappush(pq, (nextDist, there[0]))

        if start != X:
            go_party.append(dist[X])
        else:
            go_party.append(0)
            go_back = dist[1:]

    # ê°€ì¥ ì˜¤ë˜ ê±¸ë¦¬ëŠ” í•™ìƒ ì°¾ê¸°
    max_value = 0
    for i in range(N):
        max_value = max(max_value, go_party[i] + go_back[i])
    return max_value


def solution():
    adj = [[] for _ in range(N+1)]
    for frm, to, cost in edges:
        adj[frm].append((to, cost))

    answer = dijkstra(adj)
    return answer


print(solution())
