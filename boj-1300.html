import math

N = int(input())
K = int(input())


def possible(mid):
    acc = 0
    for r in range(1, N+1):
        acc += min(N, math.floor(mid / r))
        if acc >= K:
            return True

    return False


def solution():
    lo, hi, ans = 1, N**2, -1

    while lo <= hi:
        mid = (lo + hi) // 2
        if possible(mid):
            ans = mid
            hi = mid - 1
        else:
            lo = mid + 1

    return ans


print(solution())
