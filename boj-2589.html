import sys
from collections import deque

R, C = list(map(int, sys.stdin.readline().split()))
board = []
for _ in range(R):
    line = list(sys.stdin.readline().strip())
    board.append(line)

dx = [0, 0, 1, -1]
dy = [1, -1, 0, 0]


def bfs(r, c):
    q = deque()
    visit = [[0]*C for _ in range(R)]

    q.append((r, c, 0))
    visit[r][c] = 1

    longest = 0
    while q:
        x, y, cost = q.popleft()
        longest = max(longest, cost)

        for i in range(4):
            nx, ny = x + dx[i], y + dy[i]
            if 0 <= nx < R and 0 <= ny < C:
                if not visit[nx][ny] and board[nx][ny] != 'W':
                    visit[nx][ny] = 1
                    q.append((nx, ny, cost+1))

    return longest


def solution():
    ans = 0
    for r in range(R):
        for c in range(C):
            if board[r][c] == 'L':
                ans = max(ans, bfs(r, c))
    return ans


print(solution())
