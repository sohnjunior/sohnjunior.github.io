damaged = list(map(int, sys.stdin.readline().split()))
extras = list(map(int, sys.stdin.readline().split()))


def solution():
    for id in extras:
        candidates = [id, id - 1, id + 1]
        for cand in candidates:
            if cand in damaged:
                damaged.remove(cand)
                break

    answer = len(damaged)
    return answer


print(solution())
