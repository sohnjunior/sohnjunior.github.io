def move_blocks(board, dir):
    push_blocks(board, dir)

    if dir == 0:
        for r in range(N):
            # ì¸ì ‘í•œ ë‘ ë¸”ë¡ì”© ê²€ì‚¬

def max_block(board):
    ret = 0
    for r in board:
        for c in r:
            ret = max(ret, c)
    return ret


def dfs(board, depth):
    if depth == 5:
        return max_block(board)

    ret = 0
    for i in range(4):
        temp = copy.deepcopy(board)
        move_blocks(temp, i)
        ret = max(ret, dfs(temp, depth+1))
    return ret


def solution():
    answer = dfs(board, 0)
    return answer
