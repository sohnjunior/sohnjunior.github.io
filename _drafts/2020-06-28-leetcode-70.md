## 코드

~~~ javascript

/**
 * @param {number} n
 * @return {number}
 */
let memo = new Array(46).fill(-1);

var climbStairs = function (n) {
  if (n === 0) {
    return 1;
  } else if (n < 0) {
    return 0;
  }

  if(memo[n] !== -1) {
    return memo[n];
  }

  memo[n] = climbStairs(n - 1) + climbStairs(n - 2);
  return memo[n];
};

~~~
