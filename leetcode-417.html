/**
 * @param {number[][]} matrix
 * @return {number[][]}
 */
var pacificAtlantic = function (matrix) {
  if (!matrix.length) return [];

  const M = matrix.length;
  const N = matrix[0].length;

  // ê° ì§€ì ì„ íƒìƒ‰í•´ë³¸ë‹¤
  const ans = [];
  for (let x = 0; x < M; x++) {
    for (let y = 0; y < N; y++) {
      if (bfs(matrix, x, y, M, N)) {
        ans.push([x, y]);
      }
    }
  }

  return ans;
};

const dx = [0, 0, 1, -1];
const dy = [1, -1, 0, 0];

function bfs(matrix, x, y, M, N) {
  const q = [];
  const visit = new Array(M).fill(null).map((_) => new Array(N).fill(0));

  q.push([x, y]);
  visit[x][y] = 1;

  let pacific = false;
  let atlantic = false;

  while (q.length) {
    let [x, y] = q.shift();

    for (let i = 0; i < 4; i++) {
      const nx = x + dx[i],
        ny = y + dy[i];

      // pacific, atlantic, in-cell
      if (nx < 0 || ny < 0) {
        pacific = true;
      } else if (nx >= M || ny >= N) {
        atlantic = true;
      } else if (!visit[nx][ny] && matrix[x][y] >= matrix[nx][ny]) {
        visit[nx][ny] = 1;
        q.push([nx, ny]);
      }
    }
  }

  return pacific && atlantic;
}
